5. Sample Code Structure

import { BrowserRouter as Router, Route, Switch } from 'react-router-dom';
import Navbar from './components/Navbar';
import Home from './pages/Home';
import Composed from './pages/Composed';
import Gigs from './pages/Gigs';
import Film from './pages/Film';
import Artifice from './pages/Artifice';
import MusicArchive from './pages/MusicArchive';
import Team from './pages/Team';
import Production from './pages/Production';
import Forum from './pages/Forum';
import Footer from './components/Footer';

function App() {
  return (
    <Router>
      <Navbar />
      <Switch>
        <Route exact path="/" component={Home} />
        <Route path="/composed" component={Composed} />
        <Route path="/gigs" component={Gigs} />
        <Route path="/film" component={Film} />
        <Route path="/artifice" component={Artifice} />
        <Route path="/music-archive" component={MusicArchive} />
        <Route path="/team" component={Team} />
        <Route path="/production" component={Production} />
        <Route path="/forum" component={Forum} />
      </Switch>
      <Footer />
    </Router>
  );
}

export default App;



Year-wise Roller Example (YearRoller.jsx)

import React, { useState } from 'react';

const years = [2021, 2022, 2023, 2024]; // add more years as needed

function YearRoller({ onYearSelect }) {
  const [selectedYear, setSelectedYear] = useState(years[0]);

  return (
    <div className="flex overflow-x-auto space-x-4 py-2">
      {years.map(year => (
        <button
          key={year}
          className={`px-4 py-2 rounded ${selectedYear === year ? 'bg-blue-500 text-white' : 'bg-gray-200'}`}
          onClick={() => {
            setSelectedYear(year);
            onYearSelect(year);
          }}
        >
          {year}
        </button>
      ))}
    </div>
  );
}

export default YearRoller;
Use this YearRoller in your Composed.jsx to filter compositions.
